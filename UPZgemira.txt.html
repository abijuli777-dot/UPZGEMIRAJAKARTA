// Kalkulator Zakat
function calcMaal() {
var wealth = parseFloat(document.getElementById('maal-wealth').value) || 0;
var zakat = wealth * 0.025;
document.getElementById('maal-result').textContent = 'Zakat Maal 2.5% = Rp ' + zakat.toLocaleString('id-ID');
}
function calcProfesi() {
var income = parseFloat(document.getElementById('profesi-income').value) || 0;
var zakat = income * 0.025;
document.getElementById('profesi-result').textContent = 'Zakat Profesi 2.5% = Rp ' + zakat.toLocaleString('id-ID');
}
function calcFitrah() {
var people = parseFloat(document.getElementById('fitrah-people').value) || 0;
var price = parseFloat(document.getElementById('fitrah-price').value) || 0;
var zakat = people * 2.5 * price;
document.getElementById('fitrah-result').textContent = 'Total Zakat Fitrah = Rp ' + zakat.toLocaleString('id-ID');
}
function calcFidyah() {
var days = parseFloat(document.getElementById('fidyah-days').value) || 0;
var price = parseFloat(document.getElementById('fidyah-price').value) || 0;
var zakat = days * price;
document.getElementById('fidyah-result').textContent = 'Total Fidyah = Rp ' + zakat.toLocaleString('id-ID');
}
// Animasi scroll reveal
window.addEventListener('scroll', reveal);
function reveal() {
var reveals = document.querySelectorAll('.reveal');
for (var i = 0; i < reveals.length; i++) {
var windowHeight = window.innerHeight;
var elementTop = reveals[i].getBoundingClientRect().top;
var revealPoint = 150;
if (elementTop < windowHeight - revealPoint) {
reveals[i].classList.add('active');
}
}
}
